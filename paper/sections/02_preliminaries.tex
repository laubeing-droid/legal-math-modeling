\section{Preliminaries}
\label{sec:prelim}

This section establishes the formal definitions used throughout the paper.
All definitions in Sections~\ref{ssec:prelim-horn}--\ref{ssec:prelim-aaf}
have corresponding Lean formalizations in the codebase.

\subsection{Finite Sets and Monotone Operators}
\label{ssec:prelim-mono}

\begin{definition}[Finite monotone system]
\label{def:fms}
A \emph{finite monotone system} (cf.\ \texttt{FiniteMonotoneIteration.lean})
is a pair $(X, f)$ where $X$ is a finite type and $f : X \to X$ is
monotone with respect to a decidable partial order $\leq$ on $X$:
\begin{equation}
  \forall\, a\, b,\quad a \leq b \implies f(a) \leq f(b)
  \label{eq:mono-def}
\end{equation}
The \emph{iterate} of $f$ from seed $x_0$ is defined as:
\begin{equation}
  \text{iter}(f, n, x_0) =
  \begin{cases}
    x_0 & \text{if } n = 0 \\
    f(\text{iter}(f, n{-}1, x_0)) & \text{otherwise}
  \end{cases}
\end{equation}
\end{definition}

\begin{theorem}[Existence of fixed point]
\label{thm:prelim-fp}
\texttt{(exists\_fixpoint\_le\_card)} For every finite monotone system
$(X, f)$, there exists $n \leq |X|$ and a point $x^*$ such that
$f(x^*) = x^*$.
\end{theorem}

\begin{theorem}[Fixed point at cardinality bound]
\label{thm:prelim-fp-card}
\texttt{(fixed\_at\_card)} For every finite monotone system $(X, f)$,
the iterate $\text{iter}(f, |X|, x_0)$ is a fixed point of $f$ for any
seed $x_0$.
\end{theorem}

\subsection{Horn Logic}
\label{ssec:prelim-horn}

\begin{definition}[Horn clause~\cite{horn1951}]
\label{def:horn}
A \emph{Horn clause} (cf.\ \texttt{HornDefinitions.lean}) is a pair
\texttt{HornRule} with:
\begin{itemize}
  \item \texttt{premises} : a finite set of condition atoms;
  \item \texttt{conclusion} : a single head atom.
\end{itemize}
A \emph{Horn system} is a triple $(\mathcal{U}, F_0, R)$ where
$\mathcal{U}$ is a finite universe, $F_0 \subseteq \mathcal{U}$ is the
initial fact set, and $R$ is a finite set of Horn rules. The
\emph{immediate consequence operator} $T_H$ is:
\begin{equation}
  T_H(F) = F \cup \{ h \mid (h \leftarrow b_1 \wedge \cdots \wedge b_n)
  \in R,\; b_1, \ldots, b_n \in F \}
  \label{eq:TH}
\end{equation}
The \emph{Horn closure} $TH$ is the least fixpoint of $T_H$.
\end{definition}

\subsection{Abstract Argumentation Frameworks}
\label{ssec:prelim-aaf}

\begin{definition}[Dung AF~\cite{dung1995}]
\label{def:dung}
An \emph{abstract argumentation framework} (cf.\
\texttt{DungFixedPoint.lean}) is a pair $\text{DungAAF} = (A, R)$ where
$A$ is a finite set of arguments and $R \subseteq A \times A$ is an
attack relation. The \emph{characteristic function} is:
\begin{equation}
  F(S) = \{ a \in A \mid \forall b:\; (b, a) \in R \implies
  \exists c \in S:\; (c, b) \in R \}
  \label{eq:F-aaf}
\end{equation}
\end{definition}

\begin{definition}[Grounded labelling]
\label{def:grounded}
A \emph{labelling} assigns each argument to \texttt{in}, \texttt{out},
or \texttt{undecided}. The \emph{grounded specification}
(\texttt{groundedSpec}) characterizes the grounded extension as the
unique least fixed point of $F$.
\end{definition}

\subsection{Kripke Semantics}
\label{ssec:prelim-kripke}

\begin{definition}[Kripke structure~\cite{kripke1963}]
A \emph{Kripke structure} is $\mathcal{K} = (W, R, V)$ where $W$ is a
set of worlds, $R \subseteq W \times W$ is an accessibility relation,
and $V : W \to 2^{AP}$ is a valuation function. We use LTL operators
$\square$ (always) and $\Diamond$ (eventually)~\cite{pnueli1977}.
\end{definition}

\subsection{Weighted Sup-Norm Metric}
\label{ssec:prelim-metric}

\begin{definition}[Weighted sup-norm]
\label{def:wsup}
(cf.\ \texttt{WeightedSupNorm.lean}) For a finite type $\iota$ with
weight function $w : \iota \to \Real_{>0}$ and $x, y : \iota \to \Real$:
\begin{equation}
  d_w(x, y) = \sup_{i \in \iota}\; w(i) \cdot |x(i) - y(i)|
  \label{eq:wsup}
\end{equation}
\end{definition}

\subsection{Category Theory}

\begin{definition}[Category]
A \emph{category} $\mathcal{C}$ consists of objects, morphisms, and an
associative composition operator with identity
morphisms~\cite{mac1998}. A \emph{functor}
$F : \mathcal{C} \to \mathcal{D}$ preserves identity and composition.
A \emph{natural transformation} $\eta : F \Rightarrow G$ is a family
of morphisms making all naturality squares commute.
\end{definition}
