\section{Legal Ontology: $L_0$/$L_1$/$L_2$ Formalization}
\label{sec:ontology}

The juris-calculus ontology is structured in three layers of increasing
specificity, drawing on Searle's theory of institutional
facts~\cite{searle1995} and Hart's rule of recognition~\cite{hart1961}.
The ontology is encoded in the TypeScript/Lean type system; the
definitions below capture the mathematical structure.

\subsection{$L_0$: Irreducible Primitives}

\begin{definition}[$L_0$ primitive types]
The ground layer $L_0$ consists of six irreducible types:
\begin{equation}
  L_0 = \{
    \textsc{Agent},\; \textsc{Asset},\; \textsc{Act},\;
    \textsc{Status},\; \textsc{Power},\; \textsc{Defect}
  \}
  \label{eq:L0}
\end{equation}
where:
\begin{itemize}
  \item \textsc{Agent} $= \{$Seller, Buyer, Shareholder, Director, $\ldots\}$
  \item \textsc{Asset} $= \{$Goods, Shares, Patent, RealEstate, $\ldots\}$
  \item \textsc{Act} $= \{$Delivery, Payment, ShareTransfer, $\ldots\}$
  \item \textsc{Status} $= \{$Established, Valid, Pending, Voidable, Void,
        Terminated, Breached, Remedied$\}$
  \item \textsc{Power} $= \{$DispositionPower, Transferability, Alienability$\}$
  \item \textsc{Defect} $= \{$Fraud, Duress, Mistake, Illegality, Incapacity$\}$
\end{itemize}
\end{definition}

\begin{definition}[Contract validity state machine]
The \textsc{Status} type carries a directed state machine for contract
validity:
\begin{equation}
  \text{PENDING} \xrightarrow{\text{ratification}} \text{VALID}
  \xrightarrow{\text{rescission}} \text{TERMINATED}
\end{equation}
with alternative paths: $\text{VALID} \xrightarrow{\text{Abs.\ Rebuttal}}
\text{VOID}$ and $\text{VALID} \xrightarrow{\text{Cond.\ Rebuttal}}
\text{VOIDABLE} \xrightarrow{\text{rescission}} \text{VOID}$.
\end{definition}

\subsection{$L_1$: 11 Canonical Types}

\begin{definition}[$L_1$ canonical types]
The meta-ontological layer $L_1$ consists of 11 canonical legal
operation types organized into 7 enumeration domains:
\begin{equation}
  L_1 = \left\{
    \begin{array}{l}
      \text{Relationship\_Establishment},\;
      \text{Right\_Claim\_Validity},\;
      \text{Obligation\_Definition}, \\
      \text{Obligation\_Breach},\;
      \text{Remedy\_Availability},\;
      \text{Asset\_Transfer}, \\
      \text{Risk\_Allocation},\;
      \text{Defense\_Exclusion},\;
      \text{Reliance\_Principle}, \\
      \text{Legal\_Effectiveness},\;
      \text{Legal\_Stage\_Pipeline}
    \end{array}
  \right\}
  \label{eq:L1}
\end{equation}
\end{definition}

\begin{definition}[7 enumerations]
Each $L_1$ type is refined by one of 7 enumerations governing its
behavior: \textsc{Stage} (legal pipeline stage), \textsc{Effectiveness}
(valid/void/voidable), \textsc{BreachType} (material/minor/anticipatory),
\textsc{RemedyType} (damages/specific\_performance/restitution/rescission),
\textsc{TransferType} (assignment/novation/delegation),
\textsc{RiskType} (allocation/assumption/exclusion), and
\textsc{DefenseType} (affirmative/counterclaim/privilege).
\end{definition}

\begin{definition}[Legal stage pipeline]
The \textsc{Legal\_Stage\_Pipeline} defines a strict execution order:
\begin{equation}
  \text{Fact\_Finding} \to \text{Contract\_Formation} \to
  \text{Contract\_Validity} \to \text{Contract\_Interpretation} \to
  \text{Performance} \to \text{Breach} \to \text{Remedy}
  \label{eq:pipeline}
\end{equation}
Each stage must complete before the next begins.
\end{definition}

\subsection{$L_2$: Domain Concepts and DDL Modalities}

\begin{definition}[$L_2$ atom types]
Each $L_2$ concept is classified as either:
\begin{itemize}
  \item \textbf{Strict\_Atom}: a binary predicate that is either
        satisfied or not (e.g., \textsc{Delivery}, \textsc{Payment});
  \item \textbf{Defeasible\_Atom}: a predicate with override priority
        and jurisdiction-specific weight.
\end{itemize}
Each defeasible atom carries a \emph{rebuttal criteria} function:
\begin{equation}
  \text{rebut} : \text{Defeasible\_Atom} \times \text{Context} \to
  \{0, 1\} \times \text{OverridePriority}
\end{equation}
\end{definition}

\begin{definition}[DDL 4 modalities]
The Defeasible Deontic Logic (DDL) layer introduces 4 modalities for
reasoning about legal obligations with exceptions:
\begin{equation}
  \text{DDL} = \{
    \mathbf{O}(\text{obligatory}),\;
    \mathbf{P}(\text{permitted}),\;
    \mathbf{F}(\text{forbidden}),\;
    \mathbf{D}(\text{defeasibly\_entailed})
  \}
  \label{eq:ddl}
\end{equation}
The modality $\mathbf{D}$ is the key innovation: $\mathbf{D}\varphi$
means ``$\varphi$ holds unless defeated,'' capturing the rebuttable
presumption structure of legal reasoning.
\end{definition}

\begin{definition}[4 reparation modes]
When a legal obligation is breached, the remedy system selects from
4 reparation modes:
\begin{equation}
  \text{Reparation} = \{
    \textsc{Compensatory},\;
    \textsc{Restitutory},\;
    \textsc{SpecificPerformance},\;
    \textsc{Declaratory}
  \}
  \label{eq:reparation}
\end{equation}
The selection is determined by the breach type, remedy availability,
and jurisdiction-specific rules.
\end{definition}

\subsection{Cross-Jurisdiction Mapping}

\begin{definition}[Cross-jurisdiction mapping]
For each $L_2$ concept $c$, the mapping function $\mu$ assigns
jurisdiction-specific atoms:
\begin{equation}
  \mu(c) = (\mu_{\text{CN}}(c),\; \mu_{\text{HK}}(c),\; \mu_{\text{US}}(c))
\end{equation}
A concept $c$ is \emph{collision-free} if
$\mu_{\text{CN}}(c) \cap \mu_{\text{HK}}(c) \cap \mu_{\text{US}}(c)
= \emptyset$ implies no semantic overlap. Otherwise, $c$ triggers a
\emph{collision witness}.
\end{definition}
