\section{Monotone Horn Closure}
\label{sec:horn}

The Horn closure engine is the first stage of the juris-calculus
evaluator. This section presents the formal verification of its
correctness, carried out in \texttt{HornDefinitions.lean} and
\texttt{HornFixedPoint.lean}.

\subsection{Formal Setup}

\begin{definition}[Horn system]
\label{def:horn-sys}
(cf.\ \texttt{HornDefinitions.lean}) A \emph{Horn system} is a triple
$(\mathcal{U}, F_0, R)$ where:
\begin{itemize}
  \item $\mathcal{U}$ is a finite type of \emph{legal atoms};
  \item $F_0 : \mathcal{U} \to \Bool$ is the initial fact predicate;
  \item $R$ is a finite list of \texttt{HornRule}, each with
        \texttt{premises} $\subseteq \mathcal{U}$ and
        \texttt{conclusion} $\in \mathcal{U}$.
\end{itemize}
\end{definition}

\begin{definition}[$T_H$ operator]
\label{def:TH}
The \emph{immediate consequence operator} $T_H$ takes a fact predicate
$F : \mathcal{U} \to \Bool$ and returns:
\begin{equation}
  T_H(F)(a) = F(a) \;\lor\; \exists\, r \in R,\;
  r.\text{conclusion} = a \;\wedge\;
  \forall\, p \in r.\text{premises},\; F(p) = \texttt{true}
  \label{eq:TH-formal}
\end{equation}
The \emph{Horn closure} $TH$ is the least fixpoint of $T_H$.
\end{definition}

\subsection{Monotonicity}

\begin{theorem}[Horn closure is monotone]
\label{thm:horn-mono}
\texttt{(TH\_monotone)} For all fact predicates $F_1, F_2$:
\begin{equation}
  (\forall\, a,\; F_1(a) \implies F_2(a))
  \implies
  (\forall\, a,\; T_H(F_1)(a) \implies T_H(F_2)(a))
  \label{eq:horn-mono}
\end{equation}
\end{theorem}

\begin{proof}
Let $F_1 \leq F_2$ (pointwise). If $T_H(F_1)(a)$ holds, either $F_1(a)$
holds (and hence $F_2(a)$ holds by assumption), or there exists a rule
$r$ with $r.\text{conclusion} = a$ and all premises satisfied by $F_1$.
Since $F_1 \leq F_2$, the premises are also satisfied by $F_2$, so
$T_H(F_2)(a)$ holds. Verified in \texttt{HornDefinitions.lean}.
\end{proof}

\subsection{Fixpoint Theorems}

\begin{theorem}[$TH$ is a fixed point of $T_H$]
\label{thm:horn-fp}
\texttt{(horn\_result\_fixed\_point)} The Horn closure $TH$ satisfies:
\begin{equation}
  T_H(TH) = TH
  \label{eq:TH-fp}
\end{equation}
\end{theorem}

\begin{theorem}[$TH$ is the least fixed point]
\label{thm:horn-lfp}
\texttt{(horn\_result\_least\_fixed\_point)} For any $G$ satisfying
$T_H(G) = G$, we have $TH \leq G$.
\end{theorem}

\begin{proof}
By the connection to \texttt{FiniteMonotoneIteration.lean}: since
$\mathcal{U}$ is finite and $T_H$ is monotone (Theorem~\ref{thm:horn-mono}),
the finite monotone iteration theorem guarantees that the iterates
$T_H^0(F_0), T_H^1(F_0), \ldots$ reach a fixed point in at most
$|\mathcal{U}|$ steps. Minimality follows by induction on the iteration
count.
\end{proof}

\subsection{Soundness and Completeness}

\begin{theorem}[Horn soundness]
\label{thm:horn-sound}
\texttt{(horn\_soundness)} If $a \in TH$, then $a$ is derivable by a
finite chain of rule applications from $F_0$.
\end{theorem}

\begin{theorem}[Horn completeness]
\label{thm:horn-complete}
\texttt{(horn\_completeness)} If $a$ is derivable from $F_0$ by rule
applications, then $a \in TH$.
\end{theorem}

\begin{theorem}[Horn minimality]
\label{thm:horn-min}
\texttt{(horn\_result\_is\_minimal\_model)} $TH$ is the minimal model
of the Horn system: no proper subset of $TH$ satisfies all rules.
\end{theorem}

\subsection{Connection to Finite Monotone Iteration}

\begin{theorem}[Existence via finite monotone iteration]
\label{thm:horn-exists}
\texttt{(exists\_fixpoint\_le\_card, applied to Horn systems)}
The fixed point $TH$ exists and is reached in at most $|\mathcal{U}|$
iterations of $T_H$ from any seed.
\end{theorem}

\begin{remark}
The Lean formalization proves 12 theorems in the Horn module:
\texttt{TH\_monotone}, \texttt{horn\_result\_fixed\_point},
\texttt{horn\_result\_least\_fixed\_point},
\texttt{horn\_completeness}, \texttt{horn\_soundness},
\texttt{horn\_result\_is\_minimal\_model}, and 6 supporting lemmas.
No \texttt{sorry} placeholders remain in the Horn formalization.
\end{remark}
