\section{Dung AAF Grounded Extension}
\label{sec:aaf}

The second stage of the juris-calculus evaluator uses Dung's abstract
argumentation framework~\cite{dung1995} to handle rebuttal, exception,
and counter-rebuttal reasoning. This section presents the Lean
formalization in \texttt{DungFixedPoint.lean}.

\subsection{Framework Definition}

\begin{definition}[Dung AAF]
\label{def:dung-aaf}
(cf.\ \texttt{DungFixedPoint.lean}) A \texttt{DungAAF} is a pair
$(\texttt{args}, \texttt{attacks})$ where:
\begin{itemize}
  \item \texttt{args} is a finite type of arguments;
  \item \texttt{attacks} : $\texttt{args} \times \texttt{args} \to \Bool$
        is the attack relation.
\end{itemize}
The \emph{characteristic function} $F$ maps a set of accepted arguments
to the set of arguments defended against all attacks:
\begin{equation}
  F(S) = \{ a \mid \forall b,\; \text{attacks}(b, a) \implies
  \exists c \in S,\; \text{attacks}(c, b) \}
  \label{eq:F-char}
\end{equation}
\end{definition}

\begin{definition}[Grounded specification]
\label{def:grounded-spec}
The \emph{grounded specification} (\texttt{groundedSpec}) characterizes
the grounded extension $G$ as:
\begin{equation}
  \text{groundedSpec}(G) \iff
  F(G) = G \;\wedge\; \forall S,\; F(S) = S \implies G \subseteq S
  \label{eq:grounded-spec}
\end{equation}
That is, $G$ is the unique least fixed point of $F$.
\end{definition}

\subsection{Core Theorems}

\begin{theorem}[$F$ is monotone]
\label{thm:F-mono}
\texttt{(F\_monotone)} For all argument sets $S_1, S_2$:
\begin{equation}
  S_1 \subseteq S_2 \implies F(S_1) \subseteq F(S_2)
\end{equation}
\end{theorem}

\begin{theorem}[Grounded extension is a fixed point]
\label{thm:ge-fp}
\texttt{(groundedSpec\_is\_fixed\_point)} If $G$ satisfies
$\text{groundedSpec}(G)$, then $F(G) = G$.
\end{theorem}

\begin{theorem}[Grounded extension is the least fixed point]
\label{thm:ge-lfp}
\texttt{(groundedSpec\_is\_least\_fixed\_point)} If $G$ satisfies
$\text{groundedSpec}(G)$ and $S$ satisfies $F(S) = S$, then
$G \subseteq S$.
\end{theorem}

\begin{theorem}[Grounded extension is unique]
\label{thm:ge-unique}
\texttt{(groundedSpec\_unique\_least\_fixed\_point)} If both $G_1$ and
$G_2$ satisfy $\text{groundedSpec}$, then $G_1 = G_2$.
\end{theorem}

\subsection{Labelling Properties}

\begin{definition}[Three-valued labelling]
A \emph{labelling} $\mathcal{L}$ assigns each argument to one of
$\{\texttt{in}, \texttt{out}, \texttt{undecided}\}$. The grounded
extension corresponds to the set of \texttt{in}-labelled arguments.
\end{definition}

\begin{theorem}[Labelling partition]
\label{thm:lab-part}
\texttt{(labelling\_partition)} Every argument belongs to exactly one
of the three labelling classes.
\end{theorem}

\begin{theorem}[\texttt{in} soundness]
\label{thm:in-sound}
\texttt{(in\_soundness)} If $a$ is labelled \texttt{in}, then every
attacker of $a$ is labelled \texttt{out}.
\end{theorem}

\begin{theorem}[\texttt{out} soundness]
\label{thm:out-sound}
\texttt{(out\_soundness)} If $a$ is labelled \texttt{out}, then there
exists an attacker of $a$ that is labelled \texttt{in}.
\end{theorem}

\begin{theorem}[Undecided characterization]
\label{thm:undecided}
\texttt{(undecided\_characterization)} An argument $a$ is
\texttt{undecided} if and only if it is neither defended by all
\texttt{in} arguments nor attacked by an \texttt{out} argument.
\end{theorem}

\begin{theorem}[Self-attack exclusion]
\label{thm:self-attack}
\texttt{(self\_attack\_not\_in\_grounded)} If $\text{attacks}(a, a)$,
then $a$ is not in the grounded extension.
\end{theorem}
