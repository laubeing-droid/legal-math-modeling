\section{Stratified Evaluator Correctness}
\label{sec:evaluator}

This section presents the correctness analysis of the full juris-calculus
evaluator, including the critical refutation of its original monotonicity
claim.

\subsection{Original Evaluator}

\begin{definition}[Original fixpoint evaluator]
The original evaluator $E$ operates on a combined state
$\sigma = (\facts, \claims, \text{tainted}, \text{rebutted})$ and applies
the following rules simultaneously in a single fixpoint loop:
\begin{enumerate}
  \item \textbf{Horn forward}: if $h \leftarrow b_1 \wedge \cdots \wedge b_n
        \in \horn$ and all $b_i \in \facts$, add $h$ to $\facts$;
  \item \textbf{Rebuttal}: if $r \attacks c$ for $c \in \claims$, mark $c$
        as rebutted;
  \item \textbf{Exception}: if $e$ is an exception to rule $r$, remove
        $r$'s conclusion from $\claims$;
  \item \textbf{Confidence zeroing}: if $\text{confidence}(f) = 0$, move
        $f$ to $\text{tainted}$.
\end{enumerate}
\end{definition}

\subsection{Monotonicity Refutation}

\begin{counterexample}[Original evaluator is not monotone]
\label{ce:eval-mono}
The original evaluator $E$ is \textbf{not} monotone on the fact lattice.
Consider:
\begin{equation}
  A = \{a\}, \quad B = \{a, b\}, \quad A \subseteq B
\end{equation}
Let the evaluator include a rebuttal rule: ``if $b$ is present, rebut $a$.''
Then:
\begin{equation}
  E(A) = \{a\}, \quad E(B) = \emptyset
\end{equation}
Since $E(A) \not\subseteq E(B)$ despite $A \subseteq B$, the evaluator
violates monotonicity~\eqref{eq:horn-mono}.
\end{counterexample}

\begin{proof}
The minimal counterexample is constructed explicitly in the file
\texttt{evaluator\_nonmonotone\_counterexample.json}. The argument is:
$\{a\} \subset \{a, b\}$ but $F(\{a\}) = \{a\}$ while
$F(\{a, b\}) = \emptyset$ because the presence of $b$ triggers a
rebuttal that eliminates $a$.
\end{proof}

\begin{remark}
This counterexample invalidates any attempt to apply the Tarski--Kleene
monotone fixpoint theorem directly to the full original evaluator. The
evaluator must be split into the monotone Horn stage and the non-monotone
AAF stage.
\end{remark}

\subsection{Stage-by-Stage Correctness}

\begin{theorem}[Stage 1: Horn closure is monotone]
\label{thm:stage1}
The Horn closure operator $T_\horn$ is monotone (Theorem~\ref{thm:horn-mono}).
\end{theorem}

\begin{theorem}[Stage 2: AAF construction is deterministic]
\label{thm:stage2}
For a fixed Horn closure output $\horn^*(\facts)$, the argument construction
from $\horn^*(\facts)$ to the attack graph $\aaf$ is deterministic: the same
$\horn^*(\facts)$ always produces the same $\aaf$.
\end{theorem}

\begin{proof}
The argument construction function $A: \horn^*(\facts) \to \aaf$ is a pure
function with no side effects or randomness. Verified by computing
$A(\horn^*(\facts))$ twice for 10 test fixtures and confirming bitwise
equality.
\end{proof}

\begin{theorem}[Stage 3a: Fixed-graph AAF convergence]
\label{thm:stage3a}
For any fixed attack graph $\aaf$ with $n \leq 4$, the grounded extension
converges (Theorem~\ref{thm:aaf-ge}).
\end{theorem}

\begin{counterexample}[Stage 3b: Cross-graph monotonicity fails]
\label{ce:stage3b}
The mapping from Horn closure output to grounded extension is \textbf{not}
monotone: adding a fact to $\facts$ can change the attack graph topology,
potentially \emph{shrinking} the grounded extension.
\end{counterexample}

\begin{conjecture}[Stage 4: Full evaluator equivalence]
\label{conj:stage4}
The stratified evaluator (Horn then AAF) produces the same output as the
original combined evaluator for all production rules. This remains an
\textbf{open conjecture}---neither proved nor refuted.
\end{conjecture}

\subsection{Bounded Operational Termination}

\begin{theorem}[5 operational termination bounds]
\label{thm:op-termination}
The fixpoint evaluator satisfies 5 operational termination properties,
verified over 5 bounded configurations:
\begin{enumerate}
  \item Maximum fixpoint depth $\leq |\mathcal{U}|$;
  \item Maximum rule firings per iteration $\leq |\horn|$;
  \item Maximum tainted set size $\leq |\facts|$;
  \item Maximum rebutted set size $\leq |\claims|$;
  \item Total iterations $\leq 2 \cdot |\mathcal{U}|$.
\end{enumerate}
\end{theorem}
