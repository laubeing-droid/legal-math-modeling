\section{Stratified Evaluator Correctness}
\label{sec:evaluator}

This section analyzes the correctness of the full juris-calculus
evaluator, including the critical refutation of its original
monotonicity claim and the unified composition theorem.

\subsection{Original Evaluator and Non-Monotonicity}

\begin{definition}[Original fixpoint evaluator]
The original evaluator $E$ operates on a combined state
$\sigma = (\facts, \claims, \text{tainted}, \text{rebutted})$ and applies
Horn forward rules, rebuttal, exception, and confidence zeroing
simultaneously in a single fixpoint loop.
\end{definition}

\begin{counterexample}[Original evaluator is not monotone]
\label{ce:eval-mono}
The original evaluator $E$ is \textbf{not} monotone on the fact lattice.
Consider $A = \{a\} \subset B = \{a, b\}$ with a rebuttal rule ``if $b$
is present, rebut $a$'':
\begin{equation}
  E(A) = \{a\}, \quad E(B) = \emptyset
\end{equation}
Since $E(A) \not\subseteq E(B)$ despite $A \subseteq B$, the evaluator
violates monotonicity.
\end{counterexample}

\begin{remark}
This counterexample invalidates applying the Tarski--Kleene monotone
fixpoint theorem directly to the full original evaluator. The evaluator
must be split into the monotone Horn stage and the non-monotone AAF stage.
\end{remark}

\subsection{Two-Stage Stratification}

\begin{definition}[Stratified evaluator]
The \emph{stratified evaluator} separates legal reasoning into two stages:
\begin{enumerate}
  \item \textbf{Stage 1 (Horn closure)}: compute $TH(\facts)$ using
        the monotone $T_H$ operator (Section~\ref{sec:horn});
  \item \textbf{Stage 2 (AAF evaluation)}: construct the argument
        framework from $TH(\facts)$ and compute the grounded extension
        (Section~\ref{sec:aaf}).
\end{enumerate}
\end{definition}

\subsection{Cross-Graph Monotonicity Failure}

\begin{counterexample}[Cross-graph monotonicity fails]
\label{ce:stage3b}
The mapping from Horn closure output to grounded extension is \textbf{not}
monotone: adding a fact to $\facts$ can change the attack graph topology,
potentially \emph{shrinking} the grounded extension.
\end{counterexample}

\subsection{Unified Composition}

\begin{theorem}[Soundness of AAF stage]
\label{thm:soundness-aaf}
\texttt{(soundness\_aaf)} The AAF evaluation is sound: every argument
in the grounded extension is supported by the Horn closure output.
\end{theorem}

\begin{theorem}[Soundness of Banach stage]
\label{thm:soundness-banach}
\texttt{(soundness\_banach)} The Banach pricing contraction is sound:
the fixed point $\text{pricingFn}(x^*)$ is consistent with the
observed cost $T$.
\end{theorem}

\begin{theorem}[Unified composition v2]
\label{thm:unified-comp}
\texttt{(unified\_composition\_v2)} The two-stage evaluator satisfies a
commutativity property: the output of the unified pipeline equals the
sequential application of Stage 1 then Stage 2.
\end{theorem}

\begin{theorem}[Full chain]
\label{thm:full-chain}
\texttt{(full\_chain)} For any legal input satisfying the preconditions,
the full pipeline (Horn closure $\to$ AAF $\to$ output) produces a
well-formed result.
\end{theorem}

\begin{theorem}[Galois connection completeness]
\label{thm:gc2-complete}
\texttt{(gc2\_completeness)} The Galois connection between the Horn
closure lattice and the AAF output lattice preserves completeness.
\end{theorem}

\begin{theorem}[Advance preserves domain bound]
\label{thm:advance-bound}
\texttt{(advance\_preserves\_domain\_bound)} Each evaluation step
preserves the domain bound: the fact set never exceeds $\mathcal{U}$.
\end{theorem}

\begin{theorem}[Advance cannot revive refuted claims]
\label{thm:advance-no-revive}
\texttt{(advance\_cannot\_revive\_refuted)} Once a claim is refuted, no
subsequent evaluation step can restore it.
\end{theorem}

\begin{remark}
The cross-graph monotonicity failure (Counterexample~\ref{ce:stage3b})
means that full evaluator equivalence (stratified $\equiv$ original
for all production rules) remains an \textbf{open conjecture}---neither
proved nor refuted by the current formalization.
\end{remark}
