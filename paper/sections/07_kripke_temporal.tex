\section{Kripke Temporal Invariants}
\label{sec:kripke}

Legal reasoning has an intrinsic temporal dimension: facts are established
at time $t_{\text{fact}}$, procedures occur at time $t_{\text{procedure}}$,
and the temporal ordering is a legal invariant. This section formalizes the
temporal model, verified in \texttt{TemporalKripke.lean}.

\subsection{Dual-Timestamp Kripke Model}

\begin{definition}[Temporal world]
\label{def:temporal-world}
(cf.\ \texttt{TemporalKripke.lean}) A \emph{TemporalWorld} is a structure
with dual timestamps:
\begin{itemize}
  \item $t_f : W \to \Real$, the \emph{fact timestamp};
  \item $t_p : W \to \Real$, the \emph{procedure timestamp}.
\end{itemize}
A legal Kripke structure is $\mathcal{K} = (W, R, V, t_f, t_p)$ where
$(W, R, V)$ is a standard Kripke structure with the dual timestamps.
\end{definition}

\begin{definition}[Accessibility relations]
The legal Kripke model defines two distinct accessibility relations:
\begin{align}
  R_{\text{supersedes}} &= \{(w_1, w_2) \mid w_2 \text{ corrects a
    \emph{factual} error in } w_1\} \label{eq:R-super} \\
  R_{\text{corrects}} &= \{(w_1, w_2) \mid w_2 \text{ corrects a
    \emph{legal} error in } w_1\} \label{eq:R-corr}
\end{align}
\end{definition}

\subsection{Temporal Guard Invariant}

\begin{definition}[LTL temporal operators]
In Linear Temporal Logic~\cite{pnueli1977}:
\begin{itemize}
  \item $\square \varphi$ (``always $\varphi$''): $\varphi$ holds in all
        future states;
  \item $\Diamond \varphi$ (``eventually $\varphi$''): $\varphi$ holds in
        some future state.
\end{itemize}
\end{definition}

\begin{theorem}[Temporal guard always holds]
\label{thm:temporal-guard}
\texttt{(temporal\_guard\_always)} In the legal Kripke model, the
following LTL invariant holds:
\begin{equation}
  \square\,(t_f < t_p)
  \label{eq:temporal-guard}
\end{equation}
Facts are always established before the procedures that reference them.
\end{theorem}

\begin{proof}
By induction on the Kripke transition relation. The base case holds
because the initial state satisfies $t_f < t_p$. For the inductive
step, if world $w$ satisfies $t_f(w) < t_p(w)$, then every successor
$w'$ (via $R_{\text{supersedes}}$ or $R_{\text{corrects}}$) satisfies
$t_f(w') < t_p(w')$, since neither correction type can retroactively
establish a fact after its referencing procedure. Verified in
\texttt{TemporalKripke.lean}.
\end{proof}

\subsection{Litigation Guard}

\begin{theorem}[Litigation always satisfies the temporal guard]
\label{thm:lit-guard}
\texttt{(litigation\_always\_guard)} In a litigation scenario, the
temporal guard $\square(t_f < t_p)$ is preserved through all stages
of the legal pipeline (Section~\ref{sec:ontology}, equation
\eqref{eq:pipeline}).
\end{theorem}

\begin{proof}
The legal pipeline enforces strict stage ordering: fact finding
precedes contract formation, which precedes validity assessment, and
so on. Each stage transition preserves the temporal guard because
procedures in stage $k{+}1$ reference only facts established in stages
$\leq k$. Verified in \texttt{TemporalKripke.lean}.
\end{proof}

\begin{remark}
This invariant encodes a fundamental principle of procedural justice:
a court cannot reference a fact that has not yet been established. The
temporal guard prevents retroactive reasoning in the legal pipeline.
\end{remark}

\subsection{LTL Embedding}

\begin{proposition}[LTL embedding preserves reachability]
The embedding of the legal Kripke structure into LTL preserves the
accessibility relation: for all $w_1, w_2 \in W$:
\begin{equation}
  (w_1, w_2) \in R \iff \mathcal{K}, w_1 \models \Diamond\, V(w_2)
\end{equation}
This is a direct consequence of the Kripke semantics.
\end{proposition}
