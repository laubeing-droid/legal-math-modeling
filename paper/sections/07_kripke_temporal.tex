\section{Kripke Temporal Invariants}
\label{sec:kripke}

Legal reasoning has an intrinsic temporal dimension: facts are established
at time $t_{\text{fact}}$, procedures occur at time $t_{\text{procedure}}$,
and the temporal ordering is a legal invariant. This section formalizes the
temporal model.

\subsection{Dual-Timestamp Kripke Model}

\begin{definition}[Legal Kripke structure]
A \emph{legal Kripke structure} is a tuple $\kripke = (W, R, V, t_f, t_p)$
where:
\begin{itemize}
  \item $(W, R, V)$ is a standard Kripke structure;
  \item $t_f: W \to \Real$ assigns a \emph{fact timestamp} to each world;
  \item $t_p: W \to \Real$ assigns a \emph{procedure timestamp} to each
        world.
\end{itemize}
\end{definition}

\begin{definition}[Kripke accessibility relations]
The legal Kripke model defines two distinct accessibility relations:
\begin{align}
  R_{\text{supersedes}} &= \{(w_1, w_2) \mid w_2 \text{ corrects a
    \emph{factual} error in } w_1\} \label{eq:R-super} \\
  R_{\text{corrects}} &= \{(w_1, w_2) \mid w_2 \text{ corrects a
    \emph{legal} error in } w_1\} \label{eq:R-corr}
\end{align}
\end{definition}

\subsection{Mutual Exclusion}

\begin{theorem}[$R_{\text{supersedes}}$ and $R_{\text{corrects}}$ are
mutually exclusive]
\label{thm:kripke-mutex}
For the legal Kripke model:
\begin{equation}
  R_{\text{supersedes}} \cap R_{\text{corrects}} = \emptyset
  \label{eq:mutex}
\end{equation}
A state transition cannot simultaneously be a factual correction and a
legal correction.
\end{theorem}

\begin{proof}
By Z3~\cite{z3} SMT verification. The script \texttt{z3\_kripke\_mutex.py} encodes
the definitions~\eqref{eq:R-super} and~\eqref{eq:R-corr} as SMT-LIB 2
constraints and checks satisfiability of $R_{\text{supersedes}} \cap
R_{\text{corrects}} \neq \emptyset$. Result: \textsc{UNSAT}, confirming
mutual exclusion.
\end{proof}

\subsection{Temporal Guard Invariant}

\begin{definition}[LTL temporal operator]
In Linear Temporal Logic~\cite{pnueli1977,mcnaughton1993}, we use:
\begin{itemize}
  \item $\square \varphi$ (``always $\varphi$''): $\varphi$ holds in all
        future states;
  \item $\Diamond \varphi$ (``eventually $\varphi$''): $\varphi$ holds in
        some future state.
\end{itemize}
\end{definition}

\begin{theorem}[Temporal guard invariant]
\label{thm:temporal-guard}
In the legal Kripke model, the following LTL invariant holds:
\begin{equation}
  \square\,(t_f < t_p)
  \label{eq:temporal-guard}
\end{equation}
Facts are always established before the procedures that reference them.
\end{theorem}

\begin{proof}
By Z3~\cite{z3} induction. The script \texttt{z3\_temporal\_induction.py} verifies:
\begin{enumerate}
  \item \textbf{Base case}: the initial state satisfies $t_f < t_p$;
  \item \textbf{Inductive step}: if state $w$ satisfies $t_f(w) < t_p(w)$,
        then every successor $w'$ (via $R_{\text{supersedes}}$ or
        $R_{\text{corrects}}$) satisfies $t_f(w') < t_p(w')$.
\end{enumerate}
Z3 returns \textsc{UNSAT} for the negation, confirming the invariant.
\end{proof}

\begin{remark}
This invariant encodes a fundamental principle of procedural justice:
a court cannot reference a fact that has not yet been established. The
temporal guard prevents retroactive reasoning in the legal pipeline.
\end{remark}

\subsection{LTL Embedding}

\begin{proposition}[LTL embedding preserves reachability]
The embedding of the legal Kripke structure into LTL preserves the
accessibility relation: for all $w_1, w_2 \in W$:
\begin{equation}
  (w_1, w_2) \in R \iff \kripke, w_1 \models \Diamond\, V(w_2)
\end{equation}
This is a direct consequence of the Kripke semantics.
\end{proposition}
