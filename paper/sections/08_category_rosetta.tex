\section{Category-Theoretic Rosetta Mapping}
\label{sec:category}

The ``Rosetta Stone'' problem asks: can we construct a universal functor
that maps legal facts from one jurisdiction to equivalent claims in another?
This section presents the finite obstruction results from
\texttt{FiniteRosetta.lean} and the Galois connection from
\texttt{FiniteGaloisAdjunction.lean}.

\subsection{Finite Posets and Residuated Maps}

\begin{definition}[Finite poset]
\label{def:finposet}
(cf.\ \texttt{FiniteRosetta.lean}) A \texttt{FinitePoset} is a finite
type $\alpha$ with a decidable partial order $\leq$.
\end{definition}

\begin{definition}[Residuated map]
\label{def:residuated}
A \texttt{ResiduatedMap} between finite posets $P$ and $Q$ is a pair
$(f, f^*)$ where $f : P \to Q$ is monotone and $f^* : Q \to P$ is its
\emph{residual}, satisfying the adjunction property:
\begin{equation}
  f(p) \leq q \iff p \leq f^*(q)
  \label{eq:adjunction}
\end{equation}
\end{definition}

\begin{theorem}[Galois connection of residuated maps]
\label{thm:galois}
\texttt{(galois\_connection\_of\_residuated)}
(cf.\ \texttt{FiniteGaloisAdjunction.lean}) Every residuated map induces
a Galois connection between the source and target posets.
\end{theorem}

\subsection{Obstruction Theory}

\begin{definition}[Semantic-preserving functor]
A functor $F : \textbf{FactCat}_{J_1} \to \textbf{ClaimCat}_{J_2}$
is \emph{semantic-preserving} if for every morphism $f_i \to f_j$:
\begin{equation}
  F(f_i) \to F(f_j) \in \textbf{ClaimCat}_{J_2}
  \quad \text{and} \quad
  \text{legal\_meaning}(F(f_i)) \equiv \text{legal\_meaning}(f_i)
  \label{eq:semantic-preserving}
\end{equation}
\end{definition}

\begin{theorem}[No total functor exists]
\label{thm:no-total-functor}
\texttt{(no\_total\_functor)} For the toy Rosetta model, there exists
no total semantic-preserving functor $F$.
\end{theorem}

\begin{proof}
By exhaustive enumeration over all possible functor assignments on the
finite model. Zero assignments satisfy the semantic-preservation
condition~\eqref{eq:semantic-preserving} for all patterns simultaneously.
Verified in \texttt{FiniteRosetta.lean}.
\end{proof}

\begin{theorem}[Obstruction density exceeds $2/3$]
\label{thm:obstruction-density}
\texttt{(obstruction\_density\_gt\_two\_thirds)} The obstruction density
in the finite Rosetta model satisfies:
\begin{equation}
  \frac{\text{obstruction}}{\text{total}} > \frac{2}{3}
  \label{eq:obs-density}
\end{equation}
where the concrete values are $\texttt{cnOnly} = 30$,
$\texttt{collision} = 4$, and $\texttt{obstruction} = 37$.
\end{theorem}

\subsection{Collision Analysis}

\begin{definition}[Cross-jurisdiction categories]
For jurisdiction $J \in \{\text{CN}, \text{HK}, \text{US}\}$, the
\emph{fact category} $\textbf{FactCat}_J$ has objects = legal fact atoms
and morphisms = entailment relations under $J$'s legal rules. The
\emph{claim category} $\textbf{ClaimCat}_J$ is defined analogously for
legal claims.
\end{definition}

\begin{remark}
The no-total-functor result (Theorem~\ref{thm:no-total-functor}) applies
to the finite model only. Whether it extends to the full legal inventory
($> 2{,}000$ concepts) remains an open problem. The Galois connection
(Theorem~\ref{thm:galois}) provides the structural framework for any
future extension.
\end{remark}
