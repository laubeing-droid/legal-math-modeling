\section{Banach Pricing Contraction}
\label{sec:banach}

The juris-calculus pricing model estimates legal service costs using a
fixpoint iteration. This section presents the formal verification of the
contraction property for the effective-node (finite) case, carried out in
\texttt{BanachEffectiveNodes.lean}, \texttt{WeightedSupNorm.lean}, and
\texttt{ContractionCondition.lean}.

\subsection{Weighted Sup-Norm Metric}

\begin{definition}[Weighted sup-norm]
\label{def:wsup-banach}
(cf.\ \texttt{WeightedSupNorm.lean}) For a finite index type $\iota$ with
weight function $w : \iota \to \Real_{>0}$ and functions
$x, y : \iota \to \Real$:
\begin{equation}
  d_w(x, y) = \sup_{i \in \iota}\; w(i) \cdot |x(i) - y(i)|
  \label{eq:wsup-banach}
\end{equation}
\end{definition}

\begin{theorem}[Non-negativity]
\label{thm:wsup-nneg}
\texttt{(weightedSupDist\_nonneg)} For all $x, y$:
$d_w(x, y) \geq 0$.
\end{theorem}

\begin{theorem}[Triangle inequality]
\label{thm:wsup-tri}
\texttt{(weightedSupDist\_triangle)} For all $x, y, z$:
$d_w(x, z) \leq d_w(x, y) + d_w(y, z)$.
\end{theorem}

\begin{theorem}[Symmetry]
\label{thm:wsup-sym}
\texttt{(weightedSupDist\_symm)} For all $x, y$:
$d_w(x, y) = d_w(y, x)$.
\end{theorem}

\begin{theorem}[Completeness (identity of indiscernibles)]
\label{thm:wsup-complete}
\texttt{(weightedSupDist\_complete)} For all $x, y$:
$d_w(x, y) = 0 \iff x = y$.
\end{theorem}

\begin{remark}
Theorems~\ref{thm:wsup-nneg}--\ref{thm:wsup-complete} together establish
that $d_w$ is a metric on the finite function space $\iota \to \Real$.
\end{remark}

\subsection{Contraction Condition}

\begin{theorem}[Lipschitz coupling implies weighted contraction]
\label{thm:lipschitz-contraction}
\texttt{(lipschitz\_coupling\_implies\_weighted\_contraction)}
(cf.\ \texttt{ContractionCondition.lean}) If a function $f$ satisfies a
Lipschitz condition with constant $\lambda < 1$ under the weighted
sup-norm, then $f$ is a contraction mapping.
\end{theorem}

\subsection{Pricing Function}

\begin{definition}[Pricing function]
\label{def:pricing}
(cf.\ \texttt{BanachEffectiveNodes.lean}) The pricing function on the
effective-node space is:
\begin{equation}
  \text{pricingFn}(x) = \beta \cdot T + (1 - \beta) \cdot x
  \label{eq:pricing}
\end{equation}
where $x$ is the current price estimate, $T$ is the observed cost, and
$\beta \in (0, 1)$ is the smoothing parameter.
\end{definition}

\begin{theorem}[Pricing function is a contraction]
\label{thm:pricing-contraction}
\texttt{(pricingFn\_contraction)} The function $\text{pricingFn}$ is a
contraction mapping under the weighted sup-norm with contraction
constant $1 - \beta$:
\begin{equation}
  d_w(\text{pricingFn}(x), \text{pricingFn}(y))
  = (1 - \beta) \cdot d_w(x, y)
  \label{eq:pricing-contraction}
\end{equation}
\end{theorem}

\begin{proof}
By direct computation:
\begin{align}
  \text{pricingFn}(x) - \text{pricingFn}(y)
  &= (\beta T + (1{-}\beta)x) - (\beta T + (1{-}\beta)y) \notag \\
  &= (1 - \beta)(x - y)
\end{align}
Since $0 < \beta < 1$, we have $0 < 1 - \beta < 1$, confirming
contraction. The weighted sup-norm scales each component, but the
contraction factor $(1-\beta)$ is uniform. Verified in
\texttt{BanachEffectiveNodes.lean}.
\end{proof}

\begin{theorem}[Pricing function has a fixed point]
\label{thm:pricing-fp}
\texttt{(pricingFn\_fixed\_point)} The unique fixed point of
$\text{pricingFn}$ is $x^* = T$, the observed cost.
\end{theorem}

\begin{theorem}[Pricing function has a unique fixed point]
\label{thm:pricing-unique-fp}
\texttt{(pricingFn\_unique\_fixed\_point)} The fixed point $x^* = T$ is
the only point satisfying $\text{pricingFn}(x^*) = x^*$.
\end{theorem}

\begin{remark}
These results hold for the \emph{effective-node} (finite-dimensional)
case only. The full-dimensional pricing model with multiple factors
(hours, rate, complexity, jurisdiction) requires a well-defined metric
on the multi-dimensional space, which is not currently provided. See
Section~\ref{sec:conclusion} for the open problem.
\end{remark}
