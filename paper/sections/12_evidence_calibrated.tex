\section{Evidence-Calibrated Trust Label System}
\label{sec:trust}

A central contribution of this work is the \emph{evidence-calibrated trust
label system}, which tracks every mathematical claim through its proof
lifecycle and prevents unverified assertions from propagating to downstream
engineering decisions.

\subsection{Evidence Status Lattice}

\begin{definition}[7-level evidence status]
\label{def:evidence}
Every mathematical claim in juris-calculus is assigned one of seven
evidence statuses, forming a partial order $(E, \leq_E)$:
\begin{equation}
  E = \left\{
    \begin{array}{l}
      \textsc{Proved\_By\_Exhaustive\_Enumeration}, \\
      \textsc{Refuted\_By\_Counterexample}, \\
      \textsc{Partial\_Proved}, \\
      \textsc{Data\_Insufficient\_For\_Proof}, \\
      \textsc{Toy\_Synthetic\_Proof\_Only}, \\
      \textsc{Pending\_Toolchain}, \\
      \textsc{Engineering\_Baseline}
    \end{array}
  \right\}
  \label{eq:evidence-status}
\end{equation}
with the ordering:
\begin{equation}
  \textsc{Proved} >_E \textsc{Refuted} >_E \textsc{Partial}
  >_E \textsc{Data\_Insufficient} =_E \textsc{Toy}
  >_E \textsc{Pending} >_E \textsc{Engineering}
\end{equation}
\end{definition}

\begin{remark}
\textsc{Refuted} ranks above \textsc{Partial} because a refutation
(counterexample) is a \emph{definitive} result, while a partial proof
leaves the claim's truth value uncertain.
\end{remark}

\subsection{Proof Traces and Certificate Boundary}

\begin{definition}[Proof trace]
\label{def:proof-trace}
A \emph{ProofTrace} records the complete verification history of a claim:
\begin{equation}
  \text{ProofTrace} = (\text{claim\_id},\; \text{tool},\;
  \text{script\_path},\; \text{result},\; \text{timestamp},\;
  \text{lean\_file})
\end{equation}
where $\text{lean\_file}$ records the specific Lean file and theorem
name when a Lean proof exists.
\end{definition}

\begin{definition}[Certificate/checker boundary]
The system enforces a strict boundary between:
\begin{itemize}
  \item \textbf{Certificate}: the proof artifact (Lean file, Z3 script,
        Python enumeration) that establishes the claim;
  \item \textbf{Checker}: the independent verification that the certificate
        is valid (Lean kernel, Z3's proof checker, test runner).
\end{itemize}
No claim may be promoted beyond its checker-verified status.
\end{definition}

\subsection{TAINTED Fail-Closed Policy}

\begin{definition}[TAINTED marker]
A fact is marked \textsc{TAINTED} when its provenance is uncertain,
its confidence is zero, or it derives from a refuted claim. The
TAINTED marker triggers a \emph{fail-closed} policy:
\begin{equation}
  \text{TAINTED}(f) \implies f \notin \text{outputs}(\text{pipeline})
  \label{eq:tainted}
\end{equation}
A tainted fact never propagates to the final output, regardless of
whether other rules would support it.
\end{definition}

\begin{proposition}[TAINTED is monotone-incompatible]
The TAINTED fail-closed policy is non-monotone: adding a fact to the
input can cause a previously valid fact to become tainted (via rebuttal
or provenance failure). This is consistent with
Counterexample~\ref{ce:eval-mono}.
\end{proposition}

\subsection{Forbidden Tags}

\begin{definition}[Forbidden acceptance tags]
The following tags must never appear in any downstream system:
\begin{equation}
  \text{Forbidden} = \{
    \texttt{FINAL\_ALL\_THEOREMS\_PROVED},\;
    \texttt{REAL\_PRICING\_VALIDATED},\;
    \texttt{DP\_EPSILON\_LEGALLY\_DETERMINED},\;
    \texttt{ALL\_SOURCES\_OFFICIALLY\_VERIFIED}
  \end{equation}
These tags assert claims that have been either refuted or shown to be
data-insufficient.
\end{definition}

\begin{theorem}[Trust label consistency]
\label{thm:trust-consistency}
(cf.\ \texttt{JC\_Formalization.lean}, $\texttt{proved\_theorems\_card} = 7$)
The claim registry satisfies internal consistency:
\begin{enumerate}
  \item $\text{allowed}(c) \cap \text{forbidden}(c) = \emptyset$ for
        every claim $c$;
  \item $\text{status}(c) \in E$ for every claim $c$;
  \item $\text{evidence\_paths}(c) \neq \emptyset$ for every claim $c$;
  \item No duplicate claim IDs exist.
\end{enumerate}
\end{theorem}
