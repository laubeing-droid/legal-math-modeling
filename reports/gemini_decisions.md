document.addEventListener('DOMContentLoaded', function() {
    
    try {
        var topBtn = document.getElementById('ns-back-to-top');
        var bottomBtn = document.getElementById('ns-scroll-to-bottom');
        if (!topBtn && !bottomBtn) return;

        var docH = function() { return Math.max(document.body.scrollHeight, document.documentElement.scrollHeight); };
        var viewH = function() { return window.innerHeight; };
        var scrollY = function() { return Math.max(0, window.scrollY || window.pageYOffset || document.documentElement.scrollTop || 0); };
        var canScroll = function() { return (docH() - viewH()) > 80; };
        var topThreshold = function() { return Math.max(80, Math.min(180, Math.round(viewH() * 0.18))); };
        var bottomGap = function() { return Math.max(24, Math.min(80, Math.round(viewH() * 0.08))); };

        var update = function() {
            var y = scrollY();
            var maxS = Math.max(0, docH() - viewH());
            var ok = canScroll();
            if (topBtn) {
                if (ok && y >= topThreshold()) topBtn.classList.add('visible');
                else topBtn.classList.remove('visible');
            }
            if (bottomBtn) {
                if (ok && y <= maxS - bottomGap()) bottomBtn.classList.add('visible');
                else bottomBtn.classList.remove('visible');
            }
        };

        window.addEventListener('scroll', update, {passive: true});
        window.addEventListener('resize', update);
        window.addEventListener('load', update);
        if (topBtn) topBtn.addEventListener('click', function() { window.scrollTo({top: 0, behavior: 'smooth'}); });
        if (bottomBtn) bottomBtn.addEventListener('click', function() { window.scrollTo({top: docH(), behavior: 'smooth'}); });

        try {
            document.querySelectorAll('img').forEach(function(img) {
                if (!img.complete) {
                    img.addEventListener('load', update, {once: true});
                    img.addEventListener('error', update, {once: true});
                }
            });
        } catch(e) {}

        update();
        setTimeout(update, 250);
        setTimeout(update, 1000);
    } catch(e) {}

            if (!block.classList.contains('hljs')) block.classList.add('hljs');
        });

            var actions = document.createElement('div');

                        btn.classList.remove('ns-copied');
